\begin{align}
    \eta (\lambda) = \frac{\sin{\frac{\alpha + \sigma_{min} (\lambda)}{2}}}{\sin{\frac{\alpha}{2}}}

\begin{align}
    \sigma_{\eta} = 
        \left|
            \frac{
                \cos{\left( \frac{\alpha + \sigma_{min}}{2} \right)}
            }{

            }
        \right|

                        \frac{\alpha + \sigma_{min}}{2}
                    \right)
                }}{
                \tan{\left(
                    \frac{\alpha}{2}
                \right)}}
            \right)^2 \sigma_{\alpha}^2 + \sigma_{\delta}^2}
\end{align}
